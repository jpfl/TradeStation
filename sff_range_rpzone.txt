Inputs:
	RP(Numeric),
	BarSegments(Numeric);

Variables:
	BarDivision(0.0),
	lz(0.0),
	hz(0.0);

sff_range_rpzone = -1;
BarDivision = IFF(BarSegments > 0, 1.0 / BarSegments, 0);

For Value1 = 1 To BarSegments Begin
	lz = (Value1 - 1) * BarDivision;
	hz = Value1 * BarDivision;
	If RP > lz and RP <= hz Then
		sff_range_rpzone = Value1 - 1;
End;
