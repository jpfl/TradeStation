Inputs:
	RangeFactor(Numeric),
	Length(Numeric),
	RPFactor(Numeric);

Variables:
	ATR(0.0),
	MA(0.0),
	RP(0.0);

sff_pattern_xmadown = False;
ATR = Volatility(Length);
MA = Average(Close, Length);
RP = sff_range_rangepct(Close, High, Low);

If Low < MA and
High > MA and
RP <= RPFactor and
Range >= RangeFactor * ATR Then
	sff_pattern_xmadown = True;
