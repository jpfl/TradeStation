Inputs:
	PBDate(Numeric),
	PBTime(Numeric),
	PBHigh(Numeric),
	PBLow(Numeric),
	PBColor(Numeric),
	PBWidth(Numeric);

Variables:
	TL(-1);

sff_util_drawpaintbar = True;
TL = TL_New(PBDate, PBTime, PBLow, PBDate, PBTime, PBHigh);
If TL >= 0 Then Begin
	TL_SetColor(TL, PBColor);
	TL_SetStyle(TL, Tool_Solid);
	TL_SetSize(TL, PBWidth);
End Else
	sff_util_drawpaintbar = False;
