Inputs:
	Percent(Numeric),
	RangeFactor(Numeric);

Variables:
	RPOpen(0.0),
	RPClose(0.0),
	Length(20);

sff_pattern_cobra = 0;

RPOpen = sff_range_rangepct(Open, High, Low);
RPClose = sff_range_rangepct(Close, High, Low);

If RPOpen <= Percent and
RPClose >= (1 - Percent) and
Range <= RangeFactor * Average(Range, Length) Then
	sff_pattern_cobra = 1;

If RPOpen >= (1 - Percent) and
RPClose <= Percent and
Range <= RangeFactor * Average(Range, Length) Then
	sff_pattern_cobra = 2;
