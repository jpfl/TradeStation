Inputs:
	Threshold(5.0);

Variables:
	TheFloat(0.0);

If TheFloat = 0.0 Then
	TheFloat = sff_float_getfloat(GetSymbolName);

If TheFloat > 0 Then Begin
	Plot1(100 * Volume / TheFloat, "Float");
	Plot2(Threshold, "FT");
End;
