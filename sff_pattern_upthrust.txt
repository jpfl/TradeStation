Inputs:
	Strength(Numeric),
	Length(Numeric);

Variables:
	PivotBar(0),
	PivotHigh(0.0),
	RPClose(0.0);

sff_pattern_upthrust = False;
PivotBar = SwingHighBar(1, High, Strength, Length);
If PivotBar > 0 Then Begin
	PivotHigh = High[PivotBar];
	RPClose = sff_range_rangepct(Close, High, Low);
	If High > PivotHigh and
	Close < Close[1] and
	Close < Close[2] and
	Close < PivotHigh and
	Close < Open and
	RPClose < 0.5 and
	Range > Range[1] Then
		sff_pattern_upthrust = True;
End;
