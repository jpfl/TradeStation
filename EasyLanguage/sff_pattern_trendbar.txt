Inputs:
	RPO(Numeric),
	RPC(Numeric),
	DOpen(NumericSeries),
	DHigh(NumericSeries),
	DLow(NumericSeries),
	DClose(NumericSeries);

Variables:
	RPOpen(0.0),
	RPClose(0.0);

sff_pattern_trendbar = False;

RPOpen = sff_range_rangepct(DOpen, DHigh, DLow);
RPClose = sff_range_rangepct(DClose, DHigh, DLow);
Condition1 = RPOpen <= RPO and RPClose >= (1 - RPC);
Condition2 = RPOpen >= (1 - RPO) and RPClose <= RPC;

If Condition1 or Condition2 Then
	sff_pattern_trendbar = True;
