Inputs:
	Start(Numeric),
	Length(Numeric),
	Increment(Numeric),
	Offset(Numeric),
	Precision(Numeric),
	BitmapString[MaxSize](StringArrayRef);

sff_util_tag2string = False;

Value1 = Start;
For Value2 = Offset To Offset + Length - 1 Begin
	BitmapString[Value2] = NumToStr(Value1, Precision);
	Value1 = Value1 + Increment;
End;

sff_util_tag2string = True;
