Inputs:
	WedgeLength(Numeric);

Variables:
	ix(0),
	AWedge(True);

sff_pattern_wedge = False;

AWedge = True;
ix = 0;
While AWedge and ix < WedgeLength Begin
	If High[ix] > High[ix+1] or Low[ix] < Low[ix+1] Then
		AWedge = False;
	ix = ix + 1;
End;

sff_pattern_wedge = AWedge;
